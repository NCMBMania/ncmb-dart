Future<bool> delete() async {
  if (!_fields.containsKey('objectId')) {
    throw Exception('objectId is not found.');
  }
  NCMBRequest r = new NCMBRequest();
  Map res = await r.delete(_name, _fields['objectId']);
  return res.keys.length == 0;
