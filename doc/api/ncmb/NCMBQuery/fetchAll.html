Future<List> fetchAll() async {
  try {
    var r = new NCMBRequest();
    List ary = await r.get(_name, _queries);
    List results = [];
    ary.forEach((item) {
      var obj;
      switch (_name) {
        case 'files':
          obj = new NCMBFile();
          break;
        case 'users':
          obj = new NCMBUser();
          break;
        case 'roles':
          obj = new NCMBRole(item['roleName']);
          break;
        case 'installations':
          obj = new NCMBInstallation();
          break;
        case 'push':
          obj = new NCMBPush();
          break;
        default:
          obj = new NCMBObject(_name);
          break;
      }
      obj.sets(item);
      results.add(obj);
    });
    return results;
  } catch (e) {
    throw e;
  }
}
