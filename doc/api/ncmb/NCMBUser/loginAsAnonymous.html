static Future<NCMBUser> loginAsAnonymous({String id = ''}) async {
  var uuid = Uuid();
  if (id == '') {
    id = uuid.v4();
  }
  Map _fields = {
    'authData': {
      'anonymous': {'id': id}
    }
  };
  NCMBRequest r = new NCMBRequest();
  Map response = await r.post('users', _fields);
  var user = NCMBUser();
  await user.setLoginResponse(response);
  return user;
}
