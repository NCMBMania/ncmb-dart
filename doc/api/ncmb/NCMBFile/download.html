static Future<NCMBFile> download(String fileName) async {
  NCMBRequest r = new NCMBRequest();
  Map response =
      await r.exec('GET', 'files', objectId: fileName, multipart: true);
  var f = NCMBFile();
  f.data = response['data'];
  return f;
}
